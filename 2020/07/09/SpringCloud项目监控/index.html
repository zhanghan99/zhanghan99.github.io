SpringCoud项目监控

consumer中项目监控

1.导入hystrix和actuator依赖

<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-hystrix</artifactId>
    <version>1.3.5.RELEASE</version>
</dependency>
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-actuator</artifactId>
</dependency>

2.创建HystrixConfig配置类将hystrix.stream路径爆露出来

import com.netflix.hystrix.contrib.metrics.eventstream.HystrixMetricsStreamServlet;
import org.springframework.boot.web.servlet.ServletRegistrationBean;
import org.springframework.context.annotation.Bean;
import org.springframework.stereotype.Component;

@Component
public class HystrixConfig {
    @Bean
    public ServletRegistrationBean getServlet() {
        HystrixMetricsStreamServlet streamServlet = new HystrixMetricsStreamServlet();
        ServletRegistrationBean registrationBean = new ServletRegistrationBean(streamServlet);
        registrationBean.setLoadOnStartup(1);
        registrationBean.addUrlMappings("/hystrix.stream");
        registrationBean.setName("HystrixMetricsStreamServlet");
        return registrationBean;
    }
}

3.在启动类中添加注解

@EnableCircuitBreaker

4.访问hystrix.stream路径

http://localhost:8181/hystrix.stream

可视化试图监控

1.创建SpringBoot项目

2.导入SpringCloud,hystrix和actuator,dashboard,客户端依赖

<dependency>
        <groupId>org.springframework.cloud</groupId>
        <artifactId>spring-cloud-starter-hystrix</artifactId>
        <version>1.3.5.RELEASE</version>
    </dependency>
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-actuator</artifactId>
    </dependency>
    <dependency>
        <groupId>org.springframework.cloud</groupId>
        <artifactId>spring-cloud-starter-hystrix-dashboard</artifactId>
        <version>1.3.5.RELEASE</version>
    </dependency>
    <dependency>
        <groupId>org.springframework.cloud</groupId>
        <artifactId>spring-cloud-starter-netflix-eureka-client</artifactId>
    </dependency>

<dependencyManagement>
    <dependencies>
        <dependency>
            <groupId>org.springframework.cloud</groupId>
            <artifactId>spring-cloud-dependencies</artifactId>
            <version>Hoxton.SR3</version>
            <type>pom</type>
            <scope>import</scope>
        </dependency>
    </dependencies>
</dependencyManagement>

3.启动类添加注解

@EnableHystrixDashboard

4.yml文件添加配置,把项目添加到注册中心

server:
  port: 8380

eureka:
  client:
    service-url:
      defaultZone: http://root:root@127.0.0.1:9527/eureka
  instance:
    instance-id: ${spring.cloud.client.ip-address}:${server.port}
    lease-renewal-interval-in-seconds: 5
    lease-expiration-duration-in-seconds: 5
spring:
  application:
    name: wms-user-hystrix-dashboard

5.启动项目,访问

http://localhost:8380/hystrix

6.在url地址栏中输入要监控的consumer

localhost:8181/hystrix.stream

聚合监控

需要多个consumer

1.创建SpringBoot项目

2.导入SpringCloud,client,actuator,turbine依赖

<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-netflix-turbine</artifactId>
</dependency>

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-actuator</artifactId>
</dependency>

<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-netflix-eureka-client</artifactId>
</dependency>

<dependencyManagement>
    <dependencies>
        <dependency>
            <groupId>org.springframework.cloud</groupId>
            <artifactId>spring-cloud-dependencies</artifactId>
            <version>Hoxton.SR3</version>
            <type>pom</type>
            <scope>import</scope>
        </dependency>
    </dependencies>
</dependencyManagement>

3.配置yml文件,注册到服务器,并配置要监控的consumer

turbine:
  instanceUrlSuffix: hystrix.stream
  cluster-name-expression: new String('default')
  combine-host-port: true
  app-config:
  #要监控的consumer名
  WMS-USER-CONSUMER,WMS-CUSTOMER-CONSUMER
server:
  port: 8381
#放到服务器
eureka:
  client:
    service-url:
      defaultZone: http://root:root@127.0.0.1:9527/eureka
  instance:
    instance-id: ${spring.cloud.client.ip-address}:${server.port}
    lease-renewal-interval-in-seconds: 5
    lease-expiration-duration-in-seconds: 5
spring:
  application:
    name: wms-turbine

4.在启动类添加注解

//聚合监控注解
@EnableTurbine
//客户端的自动配置
@EnableEurekaClient

5.项目启动,访问可视化试图监控

http://localhost:8380/hystrix

6.输入要监控的turbien地址

http://localhost:8381/turbine.stream
